\section{Lược đồ nghiên cứu}

\begin{itemize}
    \item Nghiên cứu bổ khuyết và đánh giá hiệu quả trên các mô hình dự đoán chất lượng không khí từ nhiều nguồn dựa trên mất mát nhân tạo

\end{itemize}
